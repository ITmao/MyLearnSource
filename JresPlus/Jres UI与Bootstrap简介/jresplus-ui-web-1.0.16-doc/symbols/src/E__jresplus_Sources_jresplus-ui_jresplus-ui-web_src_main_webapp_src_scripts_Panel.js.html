 11 /**
 12  * @name Horn.Panel
 13  * @class
 14  * 面板组件</br>
 15  * 属于容器组件，一般做为单行组件的容器使用
 16  */
 17 /**@lends Horn.Panel# */
 18 /**
 19  * 组件的唯一标示
 20  * @name Horn.Panel#<b>id</b>
 21  * @type String
 22  * @default 
 23  */
 24 /**
 25  * 组件的名称
 26  * @name Horn.Panel#<b>name</b>
 27  * @type String
 28  * @default 
 29  */
 30 
 31 	Horn.Panel = Horn.extend(Horn.Base,{
 32 		COMPONENT_CLASS:"Panel",
 33 		init:function(){
 34 			 Horn.Panel.superclass.init.apply(this,arguments);
 35 			 var parent = this.el.parent('.hc_extendsearch');
 36 			 var _panel = this; 
 37 			 if(parent) {
 38 				 parent.find('.hc_btn-es-div').each(function(){
 39 					 var div = $(this);
 40 					 if(div.hasClass('hc_minus')){
 41 						 _panel.closebar = div;
 42 						div.click(function(){
 43 							_panel.hide();
 44 							//_panel.disable();    #10593查询条件被隐藏了,不应该disabled
 45 							_panel.expanedbar.show();
 46 							div.hide();
 47 						}) ;
 48 					 }else{
 49 						 _panel.expanedbar = div;
 50 						div.click(function(){
 51 							_panel.show();
 52 							//_panel.enable();    #10593
 53 							_panel.closebar.show();
 54 							div.hide();
 55 						}) ;
 56 					 }
 57 				 });
 58 			 }
 59 		},
 60         /**
 61          * 显示<br/>
 62          * 显示面板容器的内容
 63          * @name Horn.Panel#show
 64          * @function
 65          * @return {void}
 66          */
 67 		show : function(){
 68 			this.el.show();
 69 		},
 70         /**
 71          * 隐藏<br/>
 72          * 隐藏面板容器及容器内的内容
 73          * @name Horn.Panel#hide
 74          * @function
 75          * @return {void}
 76          */
 77 		hide : function(){
 78 			this.el.hide();
 79 		},
 80 		/**
 81          * 设置内部输入组件为可用状态<br/>
 82          * 对所有表单项起效（文本字段、文本区域、按钮、复选框、单选框和下拉框）
 83          * @name Horn.Panel#enable
 84          * @function
 85          * @return {void}
 86          */
 87 		enable : function(){
 88 			//this.el.find('input,textarea').not('input[realydisable],textarea[realydisable]').removeAttr('disabled');
 89 			
 90 			//BUG #6625 [panel]表单组件禁用启用
 91 			Horn.Field.findFieldCompsIn(this.el).each(function(i,f){
 92 	    		this.setDisabled(false);
 93 	    	});
 94 			//BUG #6626 [panel]禁用表单类组件
 95 			this.el.find('button,select').removeAttr('disabled');
 96 		},
 97 		/**
 98          * 设置内部输入组件为不可用状态(disabled)<br/>
 99          * 对所有表单项起效（文本字段、文本区域、按钮、复选框、单选框和下拉框）
100          * @name Horn.Panel#disable
101          * @function
102          * @return {void}
103          */
104 		disable : function(){
105 			//this.el.find('input[disabled],textarea[disabled]').attr('realydisable','yes');
106 			//this.el.find('input,textarea').not('input[realydisable],textarea[realydisable]').attr('disabled','disabled');
107 			
108 			//BUG #6625 [panel]表单组件禁用启用
109 			Horn.Field.findFieldCompsIn(this.el).each(function(i,f){
110 	    		this.setDisabled(true);
111 	    	});
112 			
113 			//BUG #6626 [panel]禁用表单类组件
114 			this.el.find('button,select').removeAttr('disabled');
115 			this.el.find('button,select').attr('disabled','disabled');
116 		}
117 	});
118 	Horn.regUI('ul.h_panel',Horn.Panel);
