304 	    			_radios.removeAttr("checked");
305 					_radios.each(function(idx,comp){

312 	    	}
313 			
314 			//设置新值之后重新校验
315 			this.validate();
316 		},

386 		isValid : function(){
387 	    	var rs = true;
388 	    	var _this = this;
389 	        var name = this.name;
390 	        if(_this.disabled == true){
391 	        	return true;
392 	        }
393 	        if(this.checkRegx.length>0){
394 	        	var val = Horn.getComp(name).getValue();
395 	        	if(val==""){
396 	        		rs = false;
397 	        		this.addMsg(_this);
398 	        		$("input[name="+name+"]").click(function(){
399 	        			_this.el.unbind('mouseover'); 
400 		        		_this.el.unbind('mouseout'); 
401 	        			_this.el.removeClass();
402 		        		_this.el.addClass("hc_radio-group");
403 	        			$("#radio_"+name).remove();
404 	        		});
405 	        		
406 	        	}else{
407 	        		this.removeMsg(_this);
408 	        	}
409 	        }
410 			return rs;
411 		},
412 		addMsg : function(_this){
413 			var userAgent = window.navigator.userAgent.toLowerCase();
414 			/*var msie10 = $.browser.msie && /msie 10\.0/i.test(userAgent);
415 	        var msie9 = $.browser.msie && /msie 9\.0/i.test(userAgent); 
416 	        var msie8 = $.browser.msie && /msie 8\.0/i.test(userAgent);*/
417 	        var msie7 = $.browser.msie && /msie 7\.0/i.test(userAgent);
418 	        var greaterThanIe7 = true;
419 	        if($.browser.msie){
420 		        //文本模式版本
421 		        if (document.documentMode && document.documentMode>=8){ // IE8文本模式
422 		        	greaterThanIe7 = true;
423 		        }else if(document.documentMode && document.documentMode<8){ // IE 5-7文本模式  
424 		        	greaterThanIe7 = false;
425 		        }
426 	        }
427 			
428 			var name = _this.name;
429 			_this.el.removeClass();
430     		_this.el.addClass("hc_radio-group hc_ver-bd");
431     		_this.el.mouseover(function(){
432     			var hcHeight = $(this).css("height");
433     			hcHeight = hcHeight.replace("px","");
434     			var tmpHeiht = parseInt(hcHeight)+3;
435     			//BUG #6912 【radio_group】在IE7下验证的错误提示信息显示的位置很远 
436     			if(!greaterThanIe7 || msie7){
437     				tmpHeiht = 0;
438     			}
439     			var msgDiv = '<span class="hc_verification" id="radio_'+name+'" style="margin-top:'+tmpHeiht+'px;">当前的单选框不能为空</span>';
440     			var msg = $("#radio_"+name);
441     			if(msg.length==0){
442     				_this.el.after(msgDiv);
443     			}else{//如果提示框已经存在，就不在创建提示框
444     				$("#radio_"+name).css("display", "inline");
445     			}
446     		});
447     		_this.el.mouseout(function(){
448     			$("#radio_"+name).css("display", "none");
449     		});
450 		},
451 		removeMsg : function(_this){
452 			var name = _this.name;
453 			_this.el.unbind('mouseover'); 
454     		_this.el.unbind('mouseout'); 
455     		_this.el.removeClass();
456     		_this.el.addClass("hc_radio-group");
457     		$("#radio_"+name).remove();
458 		},
459         reset : function() {

461         },
462         setRequired : function(required) {
463         	if (required === false) {
464         		this.setNotRequired();

478         },
479         removeRule : function(rule) {
480             var _radiogroup = this;
481             _radiogroup.el.removeClass('hc_ver-bd');
482             this.checkRegx = [];
483             var li = this.el.parent();
484             var span = $("span", li);
485             var red = $("b.hc_red", span);
486             red.html("");
487             _radiogroup.err = false;
488             this.removeMsg(this);
489         },
490         addRule : function(rule) {
491         	var _radiogroup = this;
492         	if(rule && rule.indexOf(Horn.Validate.REQUIRED) > -1){
493             	_radiogroup.err = true;
494             	this.checkRegx =[Horn.Validate.REQUIRED];
495                 var li = _radiogroup.el.parent();
496                 var span = $("span", li);
497                 var red = $("b.hc_red", span);
498                 if (!red.length) {
499                     red = $("<b>", {
500                         "class" : "hc_red",
501                         "html" : "*"
502                     });
503                     span.prepend(red);
504                 } else {
505                     red.html("*");
506                 }
507         	}else{
508         		return;
509         	}
510         }
