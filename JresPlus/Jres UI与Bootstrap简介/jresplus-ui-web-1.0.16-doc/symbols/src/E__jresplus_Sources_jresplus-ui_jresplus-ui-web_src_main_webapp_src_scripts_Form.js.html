 83 Horn.Form = Horn.extend(Horn.Base,{
 84 	COMPONENT_CLASS:"Form",
 85     form : null,
 86     /**
 87      * @ignore
 88      */
 89     ajax : false,
 90     /**
 91      * @ignore
 92      */
 93     ajaxCallback : false,
 94     /**
 95      * @ignore
 96      */
 97     onSuccessObj : null,
 98     /**
 99      * @ignore
100      */
101     baseParams:null,
102     /**
103      * @ignore
104      */
105     init : function(){
106         Horn.Form.superclass.init.apply(this,arguments) ;
107         this.form = this.el ;
108         var ajax = this.form.attr('ajax');
109         var _form = this;
110         this.baseParams = {};
111         //注册beforeSubmit事件
112 //        var beforeSubmit = this.el.attr('beforeSubmit');
113         this.beforeSubmit = this.params.beforeSubmit;
114         var _this = this;
115         $.each(this.params.events || [], function(i, o){
116         	_this[o.event] = o["function"];
117         });
118 		if(this.beforeSubmit){
119 			var beforeSubmitObj = Horn.Util.getFunObj(this.beforeSubmit);
120 		    if(beforeSubmitObj && $.type(beforeSubmitObj.fn) == "function"){
121 	           this.on('beforesubmit',beforeSubmitObj.fn);
122 	        }
123 		}
124         var submitfn = function(){
125         	return _form.fire('beforesubmit');
126         };
127 		var fn = _form.toAjaxForm;
128         _form.toAjaxForm = function(){
129         	_form.el.unbind('submit',submitfn);
130         	fn.call(_form);
131         };
132         if(!ajax) { 
133 	        _form.el.submit(submitfn);
134         }else{
135         	try{
136 	        	var ajaxObj = Horn.Util.decode(ajax);
137 	        	if($.isPlainObject(ajaxObj)){
138 	        		//注册onSuccess事件
139 			        var onSuccess = ajaxObj.onSuccess;
140 			        if(onSuccess){
141 				        var onSuccessObj = Horn.Util.getFunObj(onSuccess);
142 					    if(onSuccessObj && $.type(onSuccessObj.fn) == "function"){
143 				           this.on('ajaxsuccess',onSuccessObj.fn);
144 				        }
145 			        }
146 	        	}
147         	}catch(e){
148         	
149         	}
150         	_form.toAjaxForm();
151         }
152         _form.on('beforesubmit',_form.isValid);
153         
154         this.form.bind('reset',function(){
155         	_form.reset(false);
156         	return false;
157         });
158         
159         //若存在type为file的form组件，则改变form的媒体类型。
160         if(_form.el.find(':file').get(0)){
161         	_form.el.attr('enctype','multipart/form-data');
162         }
163     },
164     customEvents : "beforesubmit",
165     getEventTarget : function() {
166     	return this.el;
167     },
168     /**
169      * @description 调用此方法会将一个同步提交的form转换成ajaxform提交。
170      * @name Horn.Form#toAjaxForm
171      * @function
172      * @ignore
173      */
174     toAjaxForm : function(){
175     	var _form = this;
176     	if(this.ajax === true) return;
177     	this.ajax = true;
178     	this.el.submit(function(){
179     		if( _form.fire('beforesubmit') == true ){
180     			_form.doPost();
181     		}
182 			return false;
183     	});
184     } ,
185     /**
186      * @description 设置form提交时的初始化参数
187      * @function
188      * @name Horn.Form#setBaseParams
189      * @param {JSON} object
190      * @ignore
191      */
192     setBaseParams : function(Object){
193     	this.baseParams = Object;
194     },
195     /**
196      * @ignore
197      */
198     doPost : function(){
199     	var values = this.getValues();
200     	var url = this.form.attr('action');
201 
202 		Horn.apply(values,this.baseParams);
203 		url = (url.indexOf("?") === -1 ? url +"?" : url+"&") + "pagelet=" + (values.pagelet || new Date().getTime());
204 		function onAjaxComplate(result){
205 			var data = eval("("+result+")") ;
206 		    	BigPipe.onArrive(data);
207 		    	BigPipe.start();
208 		    	Horn.init();
209 		}
210 		$.post(url,values,onAjaxComplate);
211     },

220     /**
221      * @description 重置form内部所有非禁用状态表单的值
222      * @function
223      * @name Horn.Form#reset
224      */
225     reset:function(flag){
226         if(!flag === false){
227         	this.form.get(0).reset();
228         }
229         Horn.Field.findFieldCompsIn(this.form).each(function(){
230         	if($.isFunction( this.reset)){
231         		this.reset();
232         	}
233         });
234         var labels = this.form.find("div.hc_label[name]") ;
235 	    labels.each(function(idx,label){
236 	    	var $lbl = $(label);
237 	    	var hlbl = Horn.getCompByEl($lbl);
238         		hlbl.setValue('');
239 	    });
240     },

242      * @description 清空form内部所有表单的值
243      * @function
244      * @name Horn.Form#clearValue
245      */
246     clearValue:function(){
247         Horn.Field.findFieldCompsIn(this.form).each(function(){
248     		this.clearValue();
249         });
250         var labels = this.form.find("div.hc_label[name]") ;
251 	    labels.each(function(idx,label){
252 	    	var $lbl = $(label);
253 	    	var hlbl = Horn.getCompByEl($lbl);
254         		hlbl.setValue('');
255 	    });
256     },
257     /**
258      * @description 获取form的值，以名值对的方式返回
259      * @function
260      * @name Horn.Form#getValues
261      * @return Object
262      * @example 如
263      * var form = Horn.getComp("form") ;
264      * form.getValues() ;
265      * 结果--
266      * {"client_id":"1000012","fund_account":"2000112"}
267      */
268     getValues : function(){
269         var serValues = this.form.serializeArray() ;
270         return Horn.Util.arr2Obj(serValues) ;
271     },
272     /**
273      * @description 获得Form中的field，返回的是一组Horn.Field对象。
274      * @function
275      * @name Horn.Form#getFieldComps
276      * @return Horn.Field对象数组
277      */
278     getFieldComps : function(){
279     	return Horn.Field.findFieldCompsIn(this.el);
280     },
281     /**
282      * @description 给form表单元素设置值
283      * @function
284      * @name Horn.Form#setValues
285      * @param {Object}|{Array} value 设置的值，
286      * @param {String} prefix 名称前缀，用于设置'client.name'这一类的值，那么前缀必须跟上"."，即为完整值为"client."才能正常使用，
287      * 对于没有前缀的表单，prefix会被忽略。
288      * @param {Boolean} clear 若未找到是否设置为空，如果只不为false则表示清空
289      * @example 如
290      * var form = Horn.getComp("form") ;
291      * form.setValues({"client_id":"1000122","fund_account":"112222221"}) ;
292      */
293     setValues : function(v,prefix,clear){
294         var values = v ;
295         if(!prefix) prefix='';
296         if($.isArray(v)){
297             values = Horn.Util.arr2Obj(v) ;
298         }
299         this.getFieldComps().each(function(idx,field){
300         	var mutiName = field.mutiName();
301         	if(mutiName){
302         		var val = {};
303         		var i = 0;
304         		$(mutiName).each(function(idx,name){
305         			val[name] = values[name.replace(new RegExp('^'+prefix+''),'')];
306         			if(val[name]) i++;
307         		});
308         		if( i != 0 ){
309 	        		field.setValue( val || '' );
310         		}else if(!(clear == false) ){
311 	        		field.setValue( '' );
312         		}
313         	}else{
314         		var val = values[field.name.replace(new RegExp('^'+prefix+''),'')] ;
315         		if(val){
316 	        		field.setValue( val );
317         		}else if(!(clear == false)){
318 	        		field.setValue('');
319         		}
320         	}
321         });
322 	    var labels = this.form.find("div.hc_label[name|hiddenName]") ;
323 	    labels.each(function(idx,label){
324 	    	var $lbl = $(label);
325 	    	var hlbl = Horn.getCompByEl($lbl);
326 	    	var val = values[hlbl.name.replace(new RegExp('^'+prefix+''),'')] ;
327     		if(val){
328         		hlbl.setValue( val );
329     		}else if(!(clear == false)){
330         		hlbl.setValue('');
331     		}
332 	    });
333     },
334     /**
335      * @description 是否通过验证，可手动调用，用于验证某个form

357      * @description 序列表表格内容为字符串，表单的数据被格式化类似于“key10=3&key11=5&key13=2&key13=3&key14=A&key15=0”的字符串，并自动将中文字符也进行了编码转换
358      * @function
359      * @name Horn.Form#serialize
360      * @return Array
361      */
362     serialize : function(){
363         return this.form.serialize() ;
364     },
365     /**
366      * @description 从form中查找对应的field。
367      * @function
368      * @name Horn.Form#findField
369      * @param {String} name
370      * @param {String} alias
371      * @return {DomElement}
372      * @ignore
373      */
374     findField : function(name, _alias){
375         var alias = _alias||"" ;
376         var field = null;
377         var form = this.el;
378         if (jQuery.type(name) == 'string') {
379             var selector = Horn.Field.getSelector(name, alias);
380             field = $(selector, form);
381             if (!alias) {
382                 field = field.not(":disabled");
383             }
384         } else {
385             field = $(name, form);
386         }
387         return field;
388     }
