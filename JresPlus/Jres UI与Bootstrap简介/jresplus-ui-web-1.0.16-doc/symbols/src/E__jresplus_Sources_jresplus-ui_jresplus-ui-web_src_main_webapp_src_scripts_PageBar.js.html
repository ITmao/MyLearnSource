121 /**
122  * @description PageBar 的显示位置，默认不配置是居中显示，设置值为"left",居左显示，设置值为"right",居右显示,设置为"center",居中显示

130 	Horn.PageBar = Horn.extend(Horn.Base,{
131 		COMPONENT_CLASS:"PageBar",
132 		/**
133          * @description 可能会使用的跳转对象名称
134 		 * @deprecated 暂时未使用
135 		 * @ignore
136 		 */
137 		targetName : null,
138 		/**
139 		 *
140          * @description 可能会使用的跳转对象
141 		 * @deprecated 暂时未使用
142 		 * @ignore
143 		 */
144 		target : null,
145         /**
146          * @description 页面跳转地址
147          * @field
148          * @name Horn.PageBar#url
149          * @default null
150          * @ignore
151          */
152 		url : null,
153 		/**
154 		 * @ignore
155 		 */
156 		init:function(dom){
157 			Horn.PageBar.superclass.init.apply(this,arguments) ;
158 			this.targetName = this.el.attr('target');
159 			this.url = this.el.attr('url');
160 			if(!this.url) this.url = window.location.href;
161 			this.initPageParams();   //点击尾页不会跳转到尾页 ,此方法需要放在initEvent之前
162 			this.initEvent();
163 			
164 		},
165 		/**
166 		 *
167          * @description 从dom对象中获取到所需要的参数
168 		 * @private
169 		 * @function
170 		 */
171 		initPageParams : function(){
172 			var formName = "";
173         	if(this.el.attr("bindformname") && this.el.attr("bindformname").length > 0){
174         		formName = this.el.attr("bindformname") ;
175         	}
176 			var el = this.el;
177 			this.page = el.attr("pageNo") || this.page; 
178 			this.pageSize = el.attr("pageSize") || this.pageSize; 
179 			this.pageCount = el.attr("pageCount") || this.pageCount; 
180 			this.pages = el.attr("pages") || this.pages;
181 			if(this.pageCount <=0||this.pages<=0){     //如果总条数为0，将动态设置跳转页和页码input框disabled，跳转按钮disabled	
182 				$('#'+formName+'_topageid').attr("disabled","disabled"); 
183 				$('#'+formName+'_topagesize').attr("disabled","disabled"); 
184 				$('#'+formName+'_pagebtn-go').attr("disabled","disabled");
185 
186 			}else{
187 				$('#'+formName+'_topageid').removeAttr("disabled"); 
188 				$('#'+formName+'_topagesize').removeAttr("disabled"); 
189 				$('#'+formName+'_pagebtn-go').removeAttr("disabled");
190 			}
191 		},
192 		/**
193 		 * 为每个按钮绑定点击事件
194 		 * @private
195 		 * @function
196 		 */
197 		initEvent : function(){
198 			var _pageBar = this;
199 			var formName = "";
200         	if(this.el.attr("bindformname") && this.el.attr("bindformname").length > 0){
201         		formName = this.el.attr("bindformname") ;
202         	}
203 			this.el.children('a').each(function(idx,it){
204 				var item = $(it),
205 					func = function(item,f,arg){
206 						item.click(function(){
207 							if(item.hasClass('h_page-dis')){
208 								return;
209 							}
210 							if(item.hasClass('h_pagebtn-next') || item.hasClass('h_pagebtn-prev')){
211 								item.addClass("h_page-dis");
212 							}
213 							f.call(_pageBar,arg);
214 						});
215 					};
216 					
217 				if(!item.hasClass('h_page-dis')){
218 					if(item.hasClass('h_pagebtn-index')){
219 						func(item,_pageBar.goPage,_pageBar.INDEX_PAGE);
220 					}else if(item.hasClass('h_pagebtn-end')){      //尾页点击
221 						func(item,_pageBar.goPage,parseInt(_pageBar.pages));
222 					}else if(item.hasClass('h_pagebtn-next')){
223 						func(item,_pageBar.nextpage);
224 					}else if(item.hasClass('h_pagebtn-prev')){
225 						func (item,_pageBar.prevpage);
226 					}else if(item.hasClass('h_page-num')){
227 						func (item,_pageBar.goPage,parseInt(item.text()));
228 					}else if(item.hasClass('h_pagebtn-go')){       //Grid分页支持动态设置每页显示条数
229 						
230 	                    item.bind('click',function(e){
231 	                    	var _topageid = document.getElementById(formName+"_topageid").value;
232 							var _topagesize = document.getElementById(formName+"_topagesize").value;
233 							var params = [];
234 		                    params.push(parseInt(_topageid),parseInt(_topagesize));
235 	                        return _pageBar.goPage.apply(_pageBar,params);
236 	                    });
237 					}
238 				 }
239 			});
240 		},
241 		/**
242          * @description 从参数中获取到对象
243 		 * @function
244          * @name Horn.PageBar#getTarget
245          * @ignore
246 		 */
247 		getTarget:function(){
248 			return this.target || Horn.getComp(this.targetName);
249 		},
250 		/**
251 		 * 首页页码
252 		 * @property
253 		 * @ignore
254 		 */
255         INDEX_PAGE:1,
256         /**
257          * 当前页码
258 		 * @property
259 		 * @ignore
260          */
261 		page : 1,
262 		/**
263 		 * 页面大小
264 		 * @property
265 		 * @ignore
266 		 */
267 		pageSize : 20,
268 		/**
269 		 * 总条数
270 		 * @property
271 		 * @ignore
272 		 */
273 		pageCount : 0,
274 		/**
275 		 * 总页数
276 		 * @property
277 		 * @ignore
278 		 */
279 		pages : 1,
280 		/**
281 		 * 下一页
282 		 * @function
283 		 * @name Horn.PageBar#<b>nextpage</b>
284 		 */
285         nextpage : function() {
286             this.goPage(parseInt(this.page) +1 );
287         },
288         /**
289          * 上一页
290          * @function
291          * @name Horn.PageBar#<b>prevpage</b>
292          */
293         prevpage : function() {
294             this.goPage(parseInt(this.page) -1);
295         },
296         /**
297          * 跳转到首页
298          * @function
299          * @name Horn.PageBar#<b>firstpage</b>
300          */
301         firstpage : function() {
302             this.goPage(this.INDEX_PAGE);
303         },
304         /*
305         refreshpage : function() {
306             this.ajaxRequest();
307         },*/
308         /**
309          * 跳转到页面
310          * @param {Number} 跳转到页码
311          * @function
312          * @ignore
313          */
314         goPage : function(page,pageSize){
315         	var bindformname = "";
316         	if(this.el.attr("bindformname") && this.el.attr("bindformname").length > 0){
317 			    bindformname = this.el.attr("bindformname") ;
318         	}
319         	if(page < this.INDEX_PAGE ){
320         		page = this.INDEX_PAGE;
321         	}
322         	if(page > this.pages){
323         		page = this.pages;
324         	}
325         	if(pageSize > this.pageCount) {
326         		pageSize = this.pageCount;
327         	}
328         	$("#"+bindformname+"_topageid").val(page);
329         	if(pageSize < 0||pageSize==null||pageSize==undefined) pageSize = this.pageSize;
330         	this.page = page;
331         	this.pageSize = pageSize;
332 			this.doJump();
333         },
334         
335         /**
336          * 开始跳转
337          * @private
338          * @ignore
339          */
340 		doJump : function(){
341 			/*
342 			 * pageBar如果绑定了form，就以post方式进行提交查询，否则get方式进行查询
343 			 * BUG #6642 【page_bar】bindFormName设置了不能再页面第一次加载的时候生效，但是事实上第一次加载的时候form内的表单是有值的
344 			 */
345 			if(this.el.attr("bindformname") && this.el.attr("bindformname").length > 0){
346 					
347 				    var bindformname = this.el.attr("bindformname") ;
348 		            this.form = Horn.getCurrent().find("form[name='"+bindformname+"']") ;
349 		            var pageParamStr = '<input type="hidden" name="index" value="'+this.page+'"><input type="hidden" name="pageNo" value="'+this.page+'"><input type="hidden" name="pageSize" value="'+this.pageSize+'"><input type="hidden" name="count" value="'+this.pageCount+'"><input type="hidden" name="pages" value="'+this.pages+'">';
350 		            this.form.append(pageParamStr);
351 		            this.form.attr("action",this.url);
352 		            this.form.submit();
353 		            
354 		            //绑定的form表单的参数
355 					/*var formData = {};
356 					formData = Horn.Util.getValues(this.form) ;
357 		            for(var key in formData){
358 		            	if(key != "sourceurl")
359 						  addParam(key,formData[key]);
360 					}*/
361 		            
362 		    }else{
363 		    	var url = this.url;
364 				function addParam(name,val){
365 					var reg = new RegExp("([?|&])"+name+"=[^&]*");
366 					if(reg.test(url)){
367 						url=url.replace(reg,'$'+'1'+name+'='+val);
368 					}else{
369 						url += (url.indexOf('?') != -1 ? '&' : '?') + name+'='+val;
370 					}
371 				}
372 				addParam("index",this.page);
373 				addParam("pageNo",this.page);
374 				addParam("pageSize",this.pageSize);
375 				addParam("count",this.pageCount);
376 				addParam("pages",this.pages);
377 		    	Horn.Util.jump(url);
378 		    }
379 		}
380 	});
381 	Horn.regUI("div.h_pages",Horn.PageBar) ;
