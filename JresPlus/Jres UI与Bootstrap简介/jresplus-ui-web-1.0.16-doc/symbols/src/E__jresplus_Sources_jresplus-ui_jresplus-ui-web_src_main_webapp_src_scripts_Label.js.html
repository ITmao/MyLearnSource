106  /**
107   * 组件的多个值时的分割符号,定义了分割符为多选,如果定义的分割为空，则用逗号分割
108   * @name Horn.Label#<b>delimiter</b>
109   * @type String
110   * @default "" 
111   * @ignore
112   * @example
113   * 无
114   */
115 Horn.Label = Horn.extend(Horn.Base,{
116 	COMPONENT_CLASS:"Label",
117 	el:null,
118 	dictName:null,
119 	staticDict:null,
120 	mutiValue : null,
121 	init:function(dom){
122 		Horn.Label.superclass.init.apply(this,arguments);
123 		var params = this.params;
124 		if(params['name'] || params['hiddenName']){
125 			this.name = params['name'] || params['hiddenName'] ;
126 		}
127 //		this.mutiValue = params.multiple ;
128 		this.mutiValue = $(dom).attr('mutivalue');
129 		
130 		this.label = this.el.prev('span');
131 		this.hidden = this.el.find('input:hidden');
132 		if(params['hiddenName']){                            //STORY #8616 
133 			this.hidden = this.el.next('input:hidden');
134 		}
135 		/*
136 		if(!this.hidden.get(0)){
137 			this.hidden = this.el.next('input:hidden');
138 		}*/
139 		this.li= this.el.parent();
140 		
141 		if(this.hidden) {
142 			this.el.parent().append(this.hidden);
143 		}
144 		
145 		var dictName = params["dictName"],
146 			staticDict = params["items"],
147 			labelName = params['labelField'] || 'label',
148 			valueName = params['valueField'] || 'value',
149 			delimiter = params['delimiter'],
150 			value = this.hidden.get(0)?this.hidden.val():this.el.html()
151 			;
152 		if(delimiter==""||this.delimiter){
153 			this.delimiter =delimiter;
154 			this.mutiValue = true;//只要定义了分隔符，皆为多选
155 		}else{
156 			this.delimiter =",";
157 		}
158 		
159 		if(staticDict){
160 			var tmpDict = {};
161 			for(var i=0;i<staticDict.length;i++){
162 				var obj = staticDict[i];
163 				tmpDict[obj[labelName]] = obj[valueName];
164 			}
165 			staticDict = tmpDict;
166 		}
167 		
168 		if(dictName){
169 			this.dictName = dictName; 
170 			staticDict = Horn.getDict(dictName);
171 		}
172 		
173 		this.staticDict = staticDict;
174 		
175 		if(value){
176 			this.setValue(  value );
177 		}
178 	},
179 	 /**
180      * @description 设置标签的值
181      * @function
182      * @ignore
183      * @name Horn.Label#setValue
184      * @param {String} val 标签的值
185      */
186 	setValue:function(value){
187 		value = $.trim(value);
188 		if(this.hidden){
189 			this.hidden.val(value);
190 		}
191 	
192 		var dictName = this.dictName,
193 			staticDict = this.staticDict
194 			;
195 		
196 		var getVal = function(val){
197 			var tmpval = "";
198 			if(staticDict){
199 				tmpval = staticDict[val]||val;
200 			}else if(dictName){
201 				var li = $('.hc_checkboxdiv[ref_target='+dictName+'_s]').find("li[key="+val+"]");
202 				tmpval = li.attr('title')||val;
203 			}else{
204 				tmpval = val;
205 			}
206 			return tmpval;
207 		};
208 		var fval = "";
209 		var overflow = false;
210 		if(!value) value="";
211 		if(this.mutiValue){
212 			var vv = [];
213 			$(value.split(this.delimiter)).each(function(idx,item){
214 				var v = getVal(item);
215 				vv.push(v||item);
216 			});
217 			fval = vv.join(this.delimiter||',');//若存在分隔符但分隔符为空，则用逗号代替
218 			overflow = true;
219 		}else{
220 			fval = getVal(value);
221 		}
222 	
223 		this.el.html(fval);
224 		this.el.attr('title',fval);
225 		if(overflow){
226 			this.el.addClass('hc_texthide');
227 			this.el.attr('title',fval);
228 		}
229 	},
230 	/**
231      * @description 取标签的值
232      * @function
233      * @ignore
234      * @name Horn.Label#getValue
235      * @return 标签的值
236      */
237 	getValue:function(){
238 		if(this.hidden.get(0)){
239 			return this.hidden.val();
240 		}
241 		return this.el.text();
242 	},
243     /**
244      * @description 设置标签的名字
245      * @function
246      * @ignore
247      * @name Horn.Label#setLabel
248      * @param {string} val 标签的名字
249      */
250 	setLabel:function(val){
251 		this.label.html(val);
252 	},
253     /**
254      * @description 取标签的名字
255      * @function
256      * @ignore
257      * @name Horn.Label#getLabel
258      * @return 标签的名字
259      */
260 	getLabel:function(){
261 		return this.label.html();
262 	},
263 	hide:function(){
264 		this.li.hide();
265 	},
266 	show:function(){
267 		this.li.show();
268 	}
269 });

271 Horn.regUI("div.hc_label",Horn.Label) ;
