450 Horn.Combox = Horn.extend(Horn.Select,{
451 	COMPONENT_CLASS:"Combox",
452 	pLabel : "",
453 	filterFlag : "",
454 	displayField : null,
455 	keyAttr : "label",
456 	pKeyAttr : "pkey",
457 	valueAttr : "value",
458 	defaultValue : "",
459 	defaultText : "",
460 	showLabel : true,
461 	/**
462 	 * @ignore
463 	 */
464 	init : function(dom) {
465 		Horn.Combox.superclass.init.apply(this,arguments) ;
466 		this.combInit() ;
467 		//设置可能的多字典切换功能
468 		var field = this.el.find('input[ref]'),
469 			keyAttr = this.el.attr('keyfield'),
470 			pkeyAttr = this.el.attr('pkeyfield'),
471 			valueAttr = this.el.attr('titlefield')
472 			;
473 		if(keyAttr) this.keyAttr = keyAttr;	
474 		if(pkeyAttr) this.pkeyAttr = pkeyAttr;	
475 		if(valueAttr) this.valueAttr = valueAttr;
476 
477 
478 		var dictName = this.params['dictName'];
479 		if(dictName){
480 			this.field.attr('ref',dictName.split(',')[0] + (this.multipleline?'_m':'_s') ) ;
481 		}
482 		this.displayField = field;
483 		this.defaultValue = this.params.value || this.hidden.val();
484 //		this.setValue(this.defaultValue,true);
485 		
486 		//BUG #6838 combo_设置为只读的任然可以选择更改 设置为只读的任然可以选中后按退格键清除
487 		if(this.params.readonly) {
488         	this.setReadonly(true);
489         }
490 	},
491 	/**
492 	 * @ignore
493 	 */
494 	combInit : function(){
495 		var refname = this.hidden.attr("refname");
496 		if (refname) {
497 			this.field.bind("change",Horn.Util.apply(this.onCombChange,this));
498 		}
499 	},
500 	/**
501 	 * @ignore
502 	 */
503 	onCombChange : function(e, val){
504 		var refname = this.hidden.attr("refname") ;
505 		var refnames = refname.split(";") ;
506 		for(var i=0;i<refnames.length;i++){
507 			var rn = refnames[i] ;
508 			if (rn) {
509 				var rns = rn.split(",") ;
510 				Horn.getComp(rns[0],rns[1]).filterByPLabel(this.hidden.val()) ;
511 			}
512 		}
513 	},
514 	changeDict : function(name){
515    		this.clearFliter();
516 		if((this.field.get(0)).getAttribute("multiple")=="true"){
517 			name+="_m";
518 		}else{
519 			name+="_s";
520 		}
521 		this.field.attr("ref",name);
522 		this.listEl = $(
523             "div.hc_checkboxdiv[ref_target='" + name + "']");
524    		this.multipleline = this.listEl.attr("multiple_line") == "true";
525    		this.setValue("");
526 	},
527 	selectFirst : function() {
528 		var lis = this.listEl.children("ul").children("li[key]");
529 		var key = "";
530 		var text = "" ;
531 		for ( var i = 0; i < lis.length; i++) {
532 			if ($(lis.get(i)).css("display") != "none") {
533 				key = $(lis.get(i)).attr("key");
534 				text = jQuery.trim($(lis.get(i)).text());
535 				if(key!="") break;
536 			}
537 		}
538 		this.setValue({"key":key,"text":text}, true);
539 	},
540 	/**
541 	 * @ignore
542 	 */
543 	getPos : function() {
544 		return {
545 			left : 0,
546 			top : 0
547 		};
548 	},
549 	/**
550 	 * @ignore
551 	 */
552 	hideAllList : function() {
553 		var listEl = this.listEl ;
554 		$("div.hc_checkboxdiv").each(function(i, o) {
555 			if (listEl.get(0) != o) {
556 				$(o).hide();
557 				$(o).data("show_name", "");
558 			}
559 		});
560 	},
561 	/**
562 	 * @ignore
563 	 */
564 	showList : function(inputEl, listEl) {
565 		var hidden = inputEl.prev() ;
566 		var filter = hidden.data("filter") ;
567 		if(filter){
568 			this[filter.name].apply(this,filter.params) ;
569 		}
570 		else{
571 			this.clearFliter(hidden) ;
572 		}
573 		Horn.Combox.superclass.showList.apply(this,arguments) ;
574 	},
575 	/**
576      * @description 根据关联项目过滤下拉列表
577      * @function
578 	 * @name Horn.Combox#filterByPLabel
579 	 * @param pLabel 父节点值
580 	 * @returns void
581 	 * @ignore
582 	 */
583 	filterByPLabel : function(pLabel) {
584 		if(this.pLabel!=pLabel){
585 			this.reset(true);
586 			this.pLabel = pLabel ;
587 			this.hidden.data("filter" ,{
588 				name : '_filterByPLabel',
589 				params : arguments 
590 			}) ;
591 		}
592 	},
593 	/**
594 	 * @ignore
595 	 */
596 	_filterByPLabel : function(pLabel,noSelectfirst) {
597 		if (!pLabel) {
598 			return false;
599 		}
600 		var ul = this.listEl.children("ul");
601 		// 先隐藏所有的li
602 		ul.children("li[key][pKey!='" + pLabel + "']").css("display",
603 				"none");
604 		ul.children("li[key][pKey='" + pLabel + "']").css("display",
605 				"block");
606 		//若是自己设置的value值，那么在此设置为原始值。
607 		if(this.defaultValue){
608 			this.setValue(this.defaultValue);
