112     init : function(){
113         Horn.Form.superclass.init.apply(this,arguments) ;
114         this.form = this.el ;
115         var ajax = this.form.attr('ajax');
116         var _form = this;
117         this.baseParams = {};
118         //注册beforeSubmit事件
119 //        var beforeSubmit = this.el.attr('beforeSubmit');
120         this.beforeSubmit = this.params.beforeSubmit;
121         var _this = this;
122         $.each(this.params.events || [], function(i, o){
123         	_this[o.event] = o["function"];
124         });
125 		if(this.beforeSubmit){
126 			var beforeSubmitObj = Horn.Util.getFunObj(this.beforeSubmit);
127 		    if(beforeSubmitObj && $.type(beforeSubmitObj.fn) == "function"){
128 	           this.on('beforesubmit',beforeSubmitObj.fn);
129 	        }
130 		}
131         var submitfn = function(){
132         	return _form.fire('beforesubmit');
133         };
134 		var fn = _form.toAjaxForm;
135         _form.toAjaxForm = function(){
136         	_form.el.unbind('submit',submitfn);
137         	fn.call(_form);
138         };
139         if(!ajax) { 
140 	        _form.el.submit(submitfn);
141         }else{
142         	try{
143 	        	var ajaxObj = Horn.Util.decode(ajax);
144 	        	if($.isPlainObject(ajaxObj)){
145 	        		//注册onSuccess事件
146 			        var onSuccess = ajaxObj.onSuccess;
147 			        if(onSuccess){
148 				        var onSuccessObj = Horn.Util.getFunObj(onSuccess);
149 					    if(onSuccessObj && $.type(onSuccessObj.fn) == "function"){
150 				           this.on('ajaxsuccess',onSuccessObj.fn);
151 				        }
152 			        }
153 	        	}
154         	}catch(e){
155         	
156         	}
157         	_form.toAjaxForm();
158         }
159         _form.on('beforesubmit',_form.isValid);
160         
161         this.form.bind('reset',function(){
162         	_form.reset(false);
163         	return false;
164         });
165         
166         //若存在type为file的form组件，则改变form的媒体类型。
167         if(_form.el.find(':file').get(0)){
168         	_form.el.attr('enctype','multipart/form-data');
169         }
170     },

181     toAjaxForm : function(){
182     	var _form = this;
183     	if(this.ajax === true) return;
184     	this.ajax = true;
185     	this.el.submit(function(){
186     		if( _form.fire('beforesubmit') == true ){
187     			_form.doPost();
188     		}
189 			return false;
190     	});

205     doPost : function(){
206     	var values = this.getValues();
207     	var url = this.form.attr('action');
208 
209 		Horn.apply(values,this.baseParams);
210 		url = (url.indexOf("?") === -1 ? url +"?" : url+"&") + "pagelet=" + (values.pagelet || new Date().getTime());

212 			var data = eval("("+result+")") ;
213 		    	BigPipe.onArrive(data);
214 		    	BigPipe.start();
215 		    	Horn.init();
216 		}
217 		$.post(url,values,onAjaxComplate);

290     /**
291      * @description 给form表单元素设置值
292      * @function
293      * @name Horn.Form#setValues
294      * @param {Object}|{Array} value 设置的值，
295      * @param {String} prefix 名称前缀，用于设置'client.name'这一类的值，那么前缀必须跟上"."，即为完整值为"client."才能正常使用，
296      * 对于没有前缀的表单，prefix会被忽略。
297      * @param {Boolean} clear 若未找到是否设置为空，如果只不为false则表示清空
298      * @example 如
299      * var form = Horn.getComp("form") ;
300      * form.setValues({"client_id":"1000122","fund_account":"112222221"}) ;
301      * (特别说明：select和combox两个组件，赋值的json格式的值必须是字符串，不支持整型，如：错误写法：form.setValues({"select_id":0}) ;，正确写法：form.setValues({"select_id":"0"}) ;)
302      */
303     setValues : function(v,prefix,clear){
304         var values = v ;
305         if(!prefix) prefix='';
306         if($.isArray(v)){
307             values = Horn.Util.arr2Obj(v) ;
308         }
309         this.getFieldComps().each(function(idx,field){
310         	var mutiName = field.mutiName();
311         	if(mutiName){
312         		var val = {};
313         		var i = 0;
314         		$(mutiName).each(function(idx,name){
315         			val[name] = values[name.replace(new RegExp('^'+prefix+''),'')];
316         			if(val[name]) i++;
317         		});
318         		if( i != 0 ){
319 	        		field.setValue( val || '' );
320         		}else if(!(clear == false) ){
321 	        		field.setValue( '' );
322         		}
323         	}else{
324         		var val = values[field.name.replace(new RegExp('^'+prefix+''),'')] ;
325         		if(val){
326 	        		field.setValue( val );
327         		}else if(!(clear == false)){
328         			//需求 #13519 [研发中心/内部需求]使用form的setValues方法赋值时，值为0时，赋值为空，建议修复
329         			if(val==0){
330         				field.setValue( val );
331         			}else{
332         				field.setValue('');
333         			}
334         		}
335         	}
336         });
337         //bug 15434 form表单中放入表单控件，调用form的setValues方法设置表单值，调用失败
338 	    //var labels = this.form.find("div.hc_label[name|hiddenName]") ;//这种写法在jquery-1.11.3-min.js版本中会报错
339         var labels = this.form.find("div.hc_label") ;
340 	    labels.each(function(idx,label){
341 	    	var $lbl = $(label);
342 	    	var hlbl = Horn.getCompByEl($lbl);
343 	    	var val = values[hlbl.name.replace(new RegExp('^'+prefix+''),'')] ;
344 	    	//需求 #18414 form 的setValues() 方法，label值放不进去
345 			hlbl.setValue( val );
346     		/*if(val){
347         		hlbl.setValue( val );
348     		}else if(!(clear == false)){
349         		hlbl.setValue('');
350     		}*/
351 	    });
352     },

397     /**
398      * @description 从form中查找对应的field。
399      * @function
400      * @name Horn.Form#findField
401      * @param {String} name
402      * @param {String} alias
403      * @return {DomElement}
404      * @ignore
405      */
406     findField : function(name, _alias){
407         var alias = _alias||"" ;
408         var field = null;
409         var form = this.el;
410         if (jQuery.type(name) == 'string') {
411             var selector = Horn.Field.getSelector(name, alias);
412             field = $(selector, form);
413             if (!alias) {
414                 field = field.not(":disabled");
415             }
416         } else {
417             field = $(name, form);
418         }
419         return field;
420     }
