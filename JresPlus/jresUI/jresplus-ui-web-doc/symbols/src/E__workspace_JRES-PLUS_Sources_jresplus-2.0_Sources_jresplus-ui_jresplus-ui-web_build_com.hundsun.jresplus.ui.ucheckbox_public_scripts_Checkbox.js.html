 98 	Horn.Checkbox = Horn.extend(Horn.Field,{
 99 		COMPONENT_CLASS:"Checkbox",
100 		checkbox : null,
101 		init : function(){
102 			Horn.Checkbox.superclass.init.apply(this,arguments);
103 			this.checkbox = this.el.find('input:checkbox');
104 		},
105 		getEventTarget : function() {
106 			return this.el.find('input:checkbox');
107 		},
108 		/**
109          * @ignore
110          * @name Horn.CheckboxGroup#initEvents
111          * @description  正则检查字符串数组 
112          * @function 初始化方法
113          * @private
114          */
115         initEvents : function(){
116             var _checkboxgroup = this;
117             var checkboxs = _checkboxgroup.el.find('input:checkbox[check*=required]');
118             if(checkboxs && checkboxs.size()>0){
119                 this.checkRegx = $(checkboxs[0]).attr("check").split(';');
120                 checkboxs.each(function(idx,checkbox){
121                     $(checkbox).bind("click",function(){
122                     	var checkedboxs = _checkboxgroup.el.children("div").children("label").find('input:checkbox[checked]');
123                         var checked = $(this).get(0).checked;
124                    	 	if (checked) {
125                         	_checkboxgroup.removeError();
126                         }else{
127                         	_checkboxgroup.showError();
128                         }
129                     })
130                 })
131             }
132         },

155             var _checkboxgroup = this;

168          * @ignore
169          * @description 删除错误信息
170          * @function
171          * @name Horn.CheckboxGroup#removeError

173         removeError : function(){
174             var _checkboxgroup = this;
175             _checkboxgroup.el.removeClass('m-verify-error');
176             _checkboxgroup.el.addClass('m-verify-success');
177             _checkboxgroup.err = false;
178         	var msg = this.msgDiv;
179         	if(msg) msg.remove();

181         },
182         /**
183          * 设置checkbox的值(若不为true和false则直接设置给提交值)
184          * @function
185          * @name Horn.Checkbox.setValue
186          * @param 0/1或true/false
187          */
188 		setValue : function(val){
189 			if(this.checkbox.get(0)){
190 				this.checkbox.get(0).checked = !!val;
191 			}
192 			if(val && $.type( val) !='boolean' ){
193 				this.setStaticValue(val);
194 			}
195 			this.checkbox.val(val);
196 		},
197 		/**
198          * 设置对应的提交值
199          * @function
200          * @name Horn.Checkbox.setStaticValue
201          * @param {String} 预备提交的值
202          * @ignore
203 		 */
204 		setStaticValue : function(val){
205 			this.checkbox.val(val);
206 		},
207 		/**
208          * @ignore
209          * @description  内容校验
210          * @function
211          * @name Horn.CheckboxGroup.validate
212          * @ignore
213          */
214 		validate : function(){
215 			var _checkboxgroup = this;
216             var checkboxs = this.el.find('input:checkbox[check*=required]');
217             if(checkboxs && checkboxs.size()>0){
218                
219                 if(_checkboxgroup.disabled === true){
220                 	_checkboxgroup.err = false;
221                 	return;
222                 }
223                 var check = this.checkbox.get(0).checked;
224                 if(!check){
225                 	_checkboxgroup.err = true;
226                 	_checkboxgroup.showError("该选项不能为空");
227                 }else{
228                 	_checkboxgroup.err = false;
229                 	_checkboxgroup.removeError();
230                 }
231             }
232         },
233         /**
234          * @description 获取checkbox的值
235          * @function
236          * @name Horn.Checkbox.getValue
237          * @return {int} 0,1
238          */
239 		getValue : function(){
240 			return this.checkbox.val();
241 		}
