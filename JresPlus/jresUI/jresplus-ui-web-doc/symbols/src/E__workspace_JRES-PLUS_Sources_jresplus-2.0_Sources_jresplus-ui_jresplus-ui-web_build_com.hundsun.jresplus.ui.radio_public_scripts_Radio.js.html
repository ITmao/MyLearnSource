274 		COMPONENT_CLASS:"RadioGroup",
275 	    radios : null ,
276 	    name : null,
277 	    alias : null,
278 	    checkRegx : [],
279 	    init : function() {
280 	        Horn.RadioGroup.superclass.init.apply(this,arguments);
281 	        this.radios = this.el.find("input[type='radio']");
282 	        this.name = this.el.attr('name');
283 	        this.alias = this.el.attr('alias')||"";
284 	        //this.defValue=(this.params.defValue != undefined)?this.params.defValue:this.getValue();
285 	        if(this.params.check && this.params.check.indexOf("required") > -1){
286 	        	this.checkRegx =[Horn.Validate.REQUIRED];
287 	        }
288             if(this.params.value) {
289             	this.setValue(this.params.value);
290             }
291 //            if(this.params.readonly) {
292 //            	this.setReadonly(true);
293 //            }
294             if(this.params.disabled) {
295             	this.setDisabled(true);
296             }
297             this.defValue = this.params.defValue || this.params.value || "";
298             var _this = this;
299             var checkboxs = this.el.children("div").children("label").find('input:radio');
300             checkboxs.each(function(idx,checkbox){
301                 $(checkbox).bind("click",function(){
302                 	var checkedboxs = _this.getValue();
303                 	var check = $(checkbox).attr(Horn.Validate.CHECK);
304                     if(checkedboxs && checkedboxs!=""){
305                     	_this.removeError();
306                     	if (check&&check.indexOf(Horn.Validate.REQUIRED) > -1) {
307                     		_this.el.addClass('m-verify-success');
308                         }
309                     }else{
310                     	_this.showError();
311                     	_this.el.removeClass('m-verify-success');
312                     }
313                 })
314 
315             });
316 	    },
317 	    getEventTarget : function() {
318 	    	return this.el.children("div").children("label").find("input[type='radio']");
319 	    },
320 	    getValue:function(){
321 	        return this.radios.filter(":checked").val() || "";
322 	    },
323 	    setValue : function(value) {
324 	    	var radioGroup = this;
325 	    	var _radios = this.radios;
326 	    	// 需求13519--radiogroup，checkboxGroup，combox控件在from中，使用Horn.getComp("name").setValues({key1;:0});赋值无法成功
327 	    	value = value.toString();
328 	    	if (value == "") {
329 	    		this.radios.removeAttr("checked");
330 	    		//BUG #6440 【form】各个表单组件的校验提示不统一
331 	    		//this.isValid();
332 	    	} else if (this.radios.filter('[value="'+value+'"]').length > 0){
333 	    		if(this.radios.filter('[value="'+value+'"]').attr("disabled")){
334 	    			setTimeout(function(){
335 	    					_radios.attr("disabled","disabled");
336 	    					_radios.removeAttr("checked");
337 	    					_radios.filter('[value="'+value+'"]').prop("checked",true);
338 	    			},10);
339 	    		}else{
340 	    			_radios.removeAttr("checked");
341 					_radios.each(function(idx,comp){
342 						if($(comp).val() == value){
343 							$(comp).prop("checked", true);
344 						}
345 					})
346 	    		}
347 		    	
348 	    	}
349 			
350 			//设置新值之后重新校验
351 			this.validate();
352 		},
353         clearValue : function() {
354         	this.setValue("");
355         },
356 	    setEnable : function(enabled) {
357 	    	var _this = this;
358 	    	this.radios.each(function(idx,comp){
359 				if (enabled === false) {
360 					$(this).attr("disabled", "disabled");
361 					//_this.removeError(_this);
362 					//_this.err = false;
363 				} else {
364 					$(this).removeAttr("disabled");
365 				}
366 			});
367 	    },

371 	    	if(disabled === true){
372 	    		_this.disabled = true;
373 	    	}else if(disabled === false){
374 	    		_this.disabled = false;
375 	    	}
376 	    	this.radios.each(function(idx,comp){
377 				if (disabled === true) {
378 					$(this).attr("disabled", "disabled");
379 					//_this.removeError(_this);
380 					//_this.err = false;
381 				} else {
382 					$(this).removeAttr("disabled");
383 				}
384 			});
385 
386 	    },

389          * @ignore
390          * @description  内容校验
391          * @function
392          * @name Horn.CheckboxGroup.validate
393          * @ignore
394          */
395         validate : function(){
396             this.isValid();
397         },
398 		isValid : function(){
399 	    	var rs = true;
400 	    	var _this = this;
401 	        var name = this.name;
402 	        if(_this.disabled == true){
403 	        	return true;
404 	        }
405 	        if(this.checkRegx.length>0){
406 	        	var val = Horn.getComp(name).getValue();
407 	        	if(val==""){
408 	        		rs = false;
409 	        		this.showError(_this);
410 	        		
411 	        	}else{
412 	        		this.removeError(_this);
413 	        	}
414 	        }
415 			return rs;
416 		},
417 		showError : function(_this){
418 	
419 	        if(!this.msgDiv){
420         		this.msgDiv = $('<div class="m-verify-tip bottom" role="m-verify-tip" style="display: none;"></div>');
421         		this.el.after(this.msgDiv);
422         	}
423             var msg = this.msgDiv;
424             msg.html("<div class=\"verify-tip-arrow\"></div><div class=\"verify-tip-inner\">当前单选选框不能为空</div>");
425             msg.css("display", "block");
426             _this.el.addClass('m-verify-error');
427             _this.err = true;
428 		},
429 		removeError : function(_this){
430 			this.el.removeClass('m-verify-error');
431 			this.err = false;
432         	var msg = this.msgDiv;
433         	if(msg) msg.remove();
434         	var lab = this.el.parent().parent(".g-unit-wrap").children(".unit-label");
435            /* var red = $("span.m-verify-symbol", lab);
436             if(red){
437             	red.remove();
438             }*/
439         	delete this.msgDiv ;
440 		},
441         reset : function() {
442             this.setValue(this.defValue);
443         },
444         addRule : function(rule) {
445         	if(rule && rule.indexOf(Horn.Validate.REQUIRED) > -1){
446                 var _this = this;
447                 var checkboxsWithRequired = _this.el.children("div").children("label").find('input:radio[check*=required]');
448                 if(checkboxsWithRequired && checkboxsWithRequired.size()>0){
449                     return;
450                 }else{
451                     _this.err = true;
452                     this.checkRegx = rule.split(';');
453                 	var checkboxs = _this.el.children("div").children("label").find('input:radio');
454                     checkboxs.each(function(idx,checkbox){
455                     	var check = $(checkbox).attr(Horn.Validate.CHECK);
456                         if (check) {
457                             if (check.indexOf(rule) > -1) {
458                                 return;
459                             }
460                             check += Horn.Validate.CHECKSEP + rule;
461                         } else {
462                             check = rule;
463                         }
464                         $(checkbox).attr(Horn.Validate.CHECK, check);
465                     });
466                     if(rule && rule.indexOf(Horn.Validate.REQUIRED) > -1){
467         	            var lab = this.el.parent().parent(".g-unit-wrap").children(".unit-label");
468         	            var red = $("span.m-verify-symbol", lab);
469         	            if (!red.length) {
470         	                red = $("<span>", {
471         	                    "class" : "m-verify-symbol",
472         	                    "html" : "*"
473         	                });
474         	                lab.prepend(red);
475         	            } else {
476         	                red.html("*");
477         	            }
478         	            
479         	        }
480         	        //this.removeError();
481         	        //this.field.removeClass('m-verify-success');
482                 }
483         	}else{
484         		return;
485         	}
486         },
487         removeRule : function(rule) {
488             var _this = this;
489         	var checkboxs = _this.el.children("div").children("label").find('input:radio');
490             checkboxs.each(function(idx,checkbox){
491             	var check = $(checkbox).attr(Horn.Validate.CHECK);
492                 if (check && check.indexOf(rule) > -1) {
493                     var checks = check.split(Horn.Validate.CHECKSEP);
494                     checks = $.grep(checks, function(c, index) {
495                         return c && c != rule;
496                     });
497                     $(checkbox).attr(Horn.Validate.CHECK, checks.join(';'));
498                 }
499             });
500             this.checkRegx = [];
501             if(rule && rule.indexOf(Horn.Validate.REQUIRED) > -1){
502             	this.el.removeClass('m-verify-success');
503             	this.el.removeClass('m-verify-error');
504 	            var lab = this.el.parent().parent(".g-unit-wrap").children(".unit-label");
505 	            var red = $("span.m-verify-symbol", lab);
506 	            if(red){
507 	            	red.remove();
508 	            }
509 	            var msg = this.msgDiv;
510 	        	if(msg) msg.remove();
511             }
512             
513         }
