 48 	        /**
 49 	         * Formats the number according to the format string.
 50 	         * <div style="margin-left:40px">examples (123456.789):
 51 	         * <div style="margin-left:10px">
 52 	         * 0 - (123456) show only digits, no precision<br>
 53 	         * 0.00 - (123456.78) show only digits, 2 precision<br>
 54 	         * 0.0000 - (123456.7890) show only digits, 4 precision<br>
 55 	         * 0,000 - (123,456) show comma and digits, no precision<br>
 56 	         * 0,000.00 - (123,456.78) show comma and digits, 2 precision<br>
 57 	         * 0,0.00 - (123,456.78) shortcut method, show comma and digits, 2 precision<br>
 58 	         * To reverse the grouping (,) and decimal (.) for international numbers, add /i to the end.
 59 	         * For example: 0.000,00/i
 60 	         * </div></div>
 61 	         * @param {Number} v The number to format.
 62 	         * @param {String} format The way you would like to format this text.
 63 	         * @return {String} The formatted number.
 64 	         * @ignore
 65 	         */
 66 	        number: function(v, format) {
 67 	            if(!format){
 68 			        return v;
 69 			    }
 70 	            v = Number(v === null || typeof v == 'boolean' ? NaN : v);
 71 	            if (isNaN(v)){
 72 	                return '';
 73 	            }
 74 			    var comma = ',',
 75 			        dec = '.',
 76 			        i18n = false,
 77 			        neg = v < 0;
 78 			
 79 			    v = Math.abs(v);
 80 			    if(format.substr(format.length - 2) == '/i'){
 81 			        format = format.substr(0, format.length - 2);
 82 			        i18n = true;
 83 			        comma = '.';
 84 			        dec = ',';
 85 			    }
 86 			
 87 			    var hasComma = format.indexOf(comma) != -1, 
 88 			        psplit = (i18n ? format.replace(/[^\d\,]/g, '') : format.replace(/[^\d\.]/g, '')).split(dec);
 89 			
 90 			    if(1 < psplit.length){
 91 			        v = v.toFixed(psplit[1].length);
 92 			    }else if(2 < psplit.length){
 93 			        throw ('NumberFormatException: invalid format, formats should have no more than 1 period: ' + format);
 94 			    }else{
 95 			        v = v.toFixed(0);
 96 			    }
 97 			
 98 			    var fnum = v.toString();
 99 			    if(hasComma){
100 			        psplit = fnum.split('.');
101 			
102 			        var cnum = psplit[0], parr = [], j = cnum.length, m = Math.floor(j / 3), n = cnum.length % 3 || 3;
103 			
104 			        for(var i = 0; i < j; i += n){
105 			            if(i != 0){
106 			                n = 3;
107 			            }
108 			            parr[parr.length] = cnum.substr(i, n);
109 			            m -= 1;
110 			        }
111 			        fnum = parr.join(comma);
112 			        if(psplit[1]){
113 			            fnum += dec + psplit[1];
114 			        }
115 			    }
116 			
117 			    return (neg ? '-' : '') + format.replace(/[\d,?\.?]+/, fnum);
118 	        },
