129  /**
130      * @description 获取值
131      * @function
132      * @name Horn.TargetSelect#getValue
133      * @return 提交到后台的值，也就是{"label":"这里输入显示名称",key:"这里输入提交后台的id"}里的key值。
134      * 
135      */
136 Horn.TargetSelect = Horn.extend(Horn.Field,{
137 	COMPONENT_CLASS:"TargetSelect",
138     field : null ,
139     hidden : null ,
140     name : null,
141     alias : null,
142     callback: null,
143     init : function(dom){
144         Horn.TargetSelect.superclass.init.apply(this,arguments) ;
145         var _this = this ;
146         this.select = this.el;
147         this.hidden = this.select.children("input[type='hidden']");
148         this.field = this.hidden.next();
149         
150         this.name = this.hidden.attr("name") ;
151         this.alias = this.hidden.attr("alias") || "" ;
152         if(this.params.afterClear){
153         	var fn = this.params.afterClear.replace("(","").replace(")","");
154         	this.callback = window[fn];
155         }
156         var _this = this;
157         this.el.children("a").bind({
158             'click' : Horn.Util.apply(_this.delVal,_this)
159         });
160        /* $("input[name='view_"+this.name+"']").focus(function(){
161         	$("input[name='view_"+_this.name+"']").blur();
162         });*/
163     },
164     /**
165      * @description 设置值
166      * @function
167      * @name Horn.TargetSelect#setValue
168      * @param {String} value 值 传入的值必须是json格式：{"label":"这里输入显示名称",key:"这里输入提交后台的id"},如：{"label":"周智星","key":"1"}
169      * 
170      */
171     setValue : function(value) {
172     	var oldvalue=this.getValue();
173     	if(value==null||value==""){
174     		this.field.val("") ;
175     		this.hidden.val("") ;
176     	}else{
177 	        if(value instanceof Object){
178 	        	this.field.val(value.label) ;
179 	        	this.hidden.val(value.key) ;
180 	        }else{
181 	        	Horn.Msg.alert("输入错误，传入的值必须是json格式：{'label':'这里输入显示名称','key':'这里输入提交后台的id'},如：{'label':'周智星','key':'1'}");
182 	            return;
183 	        }
184     	}
185     	if(value && value.key!=oldvalue){
186 	    	this.getEventTarget().trigger("change");
187     	}
188     	this.validate();
189     },
190     /**
191      * @description 获取前台显示的值
192      * @function
193      * @name Horn.TargetSelect#getText
194      * @return  {String} label 前台显示的值，也就是{"label":"这里输入显示名称",key:"这里输入提交后台的id"}里的label的值
195      * 
196      */
197     getText : function() {
198     	var val = this.field.val();
199     	return val;
200     },
201     delVal : function(){
202     	if(this.params.disabled) return;
203     	var _this = this;
204     	var val = this.field.val();
205     	if(val!=""){
206     		Horn.Msg.confirm("确认","你确定要清除该值？",function(){  
207     			_this.hidden.val("");
208     			_this.field.val("");
209     			if(_this.callback){
210     				_this.callback();
211     			}
212     		},function(){  
213     		    //canel  
214     		});
215     		
216     	}
217     }
218 }) ;
