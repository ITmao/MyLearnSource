122 		COMPONENT_CLASS:"PasswordGroup",
123 		field1 : null,
124 		field2 : null,
125 		name1 : null,
126 		name2 : null,
127 		defValue1:"",
128 		defValue2:"",
129 		init : function(dom){
130 			var inputs = $(dom).find('input');
131 			this.field1 = $(inputs[0]);
132 			this.field2 = $(inputs[1]);
133 			this.name1 = this.field1.attr('name');
134 			this.name2 = this.field2.attr('name');
135 			Horn.PasswordGroup.superclass.init.apply(this,arguments);
136 			if(this.field1){
137 				this.defValue1 = (this.params.defValue != undefined)?this.params.defValue:this.field1.val();
138 				this.defValue2 = (this.params.defValue != undefined)?this.params.defValue:this.field2.val();
139 			}
140 		},
141 		initEvents : function(){
142 	    	var _pwdgroup = this;
143 	    	this.field2.bind('blur',function(){
144 		    		
145 	    		var val2 =_pwdgroup.field2.val();
146     			if(val2!=""){
147     				val2 = val2.replace(/(^\s*)|(\s*$)/g,'');
148     				_pwdgroup.field2.val(val2);
149     			}
150     			var val1 =_pwdgroup.field1.val();
151     			if(val1!=""){
152     				val1 = val1.replace(/(^\s*)|(\s*$)/g,'');
153     				_pwdgroup.field1.val(val1);
154     			}
155 	    		Horn.Validate.onValid({data:[Horn.Validate,_pwdgroup]});
156 	    		if(val1!=""&&val2!=""){
157 		    		if(_pwdgroup.isValid()){
158 			    		var pwd = _pwdgroup.field1.val();
159 			    		if(_pwdgroup.field2.val() != pwd){
160 			    			_pwdgroup.showError('校验不匹配！');
161 			    		}else{
162 			    			_pwdgroup.removeError();
163 			    		}
164 			    	}
165 	    		}else{
166 	    			var rule2 =  _pwdgroup.field2.attr("check");
167 	    			if(rule2 && rule2.indexOf(Horn.Validate.REQUIRED) > -1){
168 	    				_pwdgroup.showError("当前输入不能为空");
169 	    			}
170 	    		}
171 	    	});
172 	    	this.field1.blur(function(){
173 	    		var val2 =_pwdgroup.field2.val();
174     			if(val2!=""){
175     				val2 = val2.replace(/(^\s*)|(\s*$)/g,'');
176     				_pwdgroup.field2.val(val2);
177     			}
178     			var val1 =_pwdgroup.field1.val();
179     			if(val1!=""){
180     				val1 = val1.replace(/(^\s*)|(\s*$)/g,'');
181     				_pwdgroup.field1.val(val1);
182     			}
183     			Horn.Validate.onValid({data:[Horn.Validate,_pwdgroup]});
184     			var pwd1 = _pwdgroup.field1.val();
185 	    		var pwd2 = _pwdgroup.field2.val();
186 	    		if(pwd1!=""&&pwd2!=""){
187 		    		if(_pwdgroup.isValid()){
188 			    		if(pwd1 != pwd2 && pwd2){
189 			    			_pwdgroup.showError('校验不匹配！');
190 			    		}else{
191 			    			_pwdgroup.removeError();
192 			    		}
193 		    		}
194 	    		}else{
195 		    		var rule1 =  _pwdgroup.field1.attr("check");
196 	    			if(rule1 && rule1.indexOf(Horn.Validate.REQUIRED) > -1){
197 	    				_pwdgroup.showError("当前输入不能为空");
198 	    			}
199 	    		}
200 	    	});
201 	    },
202         /**
203          * @description 验证输入密码有效性
204          * @function
205          * @name Horn.PasswordGroup#validate
206          */
207 	    validate : function(){
208 	    	this.field1.blur();
209 	    	this.field2.blur();
210 	    },
211 	    /**
212 	     * 增加校验规则
213 	     * @function
214 	     * @name Horn.Field#addRule
215 	     * @param {String} rule 校验规则字符串
216 	     * @ignore
217 	     */
218 	    addRule : function(rule) {
219 	        var input = this.field1;
220 	        var input2 = this.field2;
221 	        var check = input.attr(Horn.Validate.CHECK);
222 	        if (check) {
223 	            if (check.indexOf(rule) > -1) {
224 	                return;
225 	            }
226 	            check += Horn.Validate.CHECKSEP + rule;
227 	        } else {
228 	            check = rule;
229 	        }
230 	        input.attr(Horn.Validate.CHECK, check);
231 	        input2.attr(Horn.Validate.CHECK, check);
232 	        if(rule && rule.indexOf(Horn.Validate.REQUIRED) > -1){
233 	            var li = this.el.parent().parent(".g-unit-wrap");
234 	            var lab = $("label", li);
235 	            var red = $("span.m-verify-symbol", lab);
236 	            if (!red.length) {
237 	                red = $("<span>", {
238 	                    "class" : "m-verify-symbol",
239 	                    "html" : "*"
240 	                });
241 	                lab.prepend(red);
242 	            } else {
243 	                red.html("*");
244 	            }
245 	            
246 	        }
247 	        this.removeError();
248 	        this.field.removeClass('m-verify-success');
249 	    },
250 	    /**
251 	     * 删除校验规则
252 	     * @function
253 	     * @name Horn.Field#removeRule
254 	     * @param {String} rule 校验规则字符串
255 	     * @ignore
256 	     */
257 	    removeRule : function(rule) {
258 	        var input = this.field1;
259 	        var input2 = this.field2;
260 	        var check = input.attr(Horn.Validate.CHECK);
261 	          //BUG #6518 【calendar】先进行非空校验的错误提示，然后调用removeRule("qq")，会造成非空校验的错误提示消失
262 	        if (check && check.indexOf(rule) > -1) {//如果要去除的在原来的验证规则了就删除，否则不删除
263 	            var checks = check.split(Horn.Validate.CHECKSEP);
264 	            checks = $.grep(checks, function(c, index) {
265 	                return c && c != rule;
266 	            });
267 	            input.attr(Horn.Validate.CHECK, checks.join(';'));
268 	            input2.attr(Horn.Validate.CHECK, checks.join(';'));
269 	            this.removeError();
270 	            this.setNotRequired();
271 	        }
272 	    },
273 	    /**
274 	     * 显示验证错误信息
275 	     * @function
276 	     * @name Horn.Field#showError
277 	     * @param {String} 错误信息
278 	     * @ignore
279 	     */
280 	    showError : function(errorMsg){
281 	    	var field1 = this.field1; 
282 	    	var field2 = this.field2;
283 	    	field1.removeClass('m-verify-success');
284 	    	field1.addClass('m-verify-error');
285 	    	field2.removeClass('m-verify-success');
286 	    	field2.addClass('m-verify-error');
287 	    	errorMsg = $.type(errorMsg) == "boolean" ? "校验错误" : errorMsg;
288 	    	if(!this.msgDiv){
289 	    		this.msgDiv = $('<div class="m-verify-tip bottom" role="m-verify-tip" style="display: none;"></div>');
290 	    		this.el.after(this.msgDiv);
291 	    	}
292 	        var msg = this.msgDiv;
293 	        msg.html("<div class=\"verify-tip-arrow\"></div><div class=\"verify-tip-inner\">"+errorMsg+"</div>");
294 	        msg.css("display", "block");
295 	        this.err = true;
296 	    },
297 	    /**
298 	     * 删除错误信息
299 	     * @function
300 	     * @name Horn.Field#removeError
301 	     * @ignore
302 	     */
303 	    removeError : function(){
304 	    	this.field1.removeClass('m-verify-error');
305 	    	this.field2.removeClass('m-verify-error');
306 	        var input = this.get();
307 	        var check = this.field1.attr(Horn.Validate.CHECK);
308 	        if (check) {
309 	        	if(this.isValid){
310 	        		if(this.field1.val()!=""||this.field2.val()!=""){
311 		            	this.field1.addClass('m-verify-success');
312 		            	this.field2.addClass('m-verify-success');
313 	        		}
314 	            }
315 	        }
316 	        this.err = false;
317 	    	var msg = this.msgDiv;
318 	    	if(msg) msg.remove();
319 	    	delete this.msgDiv ;
320 	    },

341 		},
342 
343 		
344 		/**
345          * @description 如果设置了defValue的值，重置成的defValue值，否则重置成value值
346          * @function
347          * @name Horn.PasswordGroup#reset
348          */
349 		reset : function(clear){
350     		var defValue = clear?"":this.defValue1;
351 	    	this.setValue(defValue);
352 		},

354 		 * 清空表单的值，显示值和隐藏值都设置为""
355 		 * @function
356 		 * @name Horn.PasswordGroup#clearValue
357 		 */
358 	    clearValue : function() {
359 	    	this.setValue("");
360 	    },
361 		setEnable : function(enable){
362 			var _pwdgroup = this;
363 			if(enable){
364 				this.field1.removeAttr("disabled");
365 				this.field2.removeAttr("disabled");
366 				this.field1.blur();
367 				this.field2.blur();
368 			}else{
369 				this.field1.attr("disabled","disabled");
370 				this.field2.attr("disabled","disabled");
371     			_pwdgroup.removeError();
372 			}
373 		}
374 	});
