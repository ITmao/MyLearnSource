1 /*
  2 	*** jquery.drag 容器拖拽
  3 */
  4 (function($){
  5 	var _window = {
  6         windowWidth: function() {
  7             return getMsie() ? document.documentElement.clientWidth: window.innerWidth
  8         },
  9         windowHeight: function() {
 10             return getMsie() ? document.documentElement.clientHeight: window.innerHeight
 11         },
 12         scrollTop: function() {
 13             return getMsie() ? document.documentElement.scrollTop: window.pageYOffset
 14         },
 15         scrollLeft: function() {
 16             return getMsie() ? document.documentElement.scrollLeft: window.pageXOffset
 17         }
 18 	};
 19 	function getMsie(){
 20     	var rv = false;
 21     	  if (navigator.appName == 'Microsoft Internet Explorer'){
 22     	    rv = true;
 23     	  }else if (navigator.appName == 'Netscape'){
 24     	     rv = true;
 25     	  }
 26     	return rv;
 27     }
 28 	// 清除文本选择
 29 	function clsSelect(e){
 30 		e = e || window.event;
 31 		if('getSelection' in window){
 32 			window.getSelection().removeAllRanges();
 33 		}else{
 34 			try {
 35 				document.selection.empty();
 36 			} catch (e) {};
 37 		};
 38 		if(e.preventDefault){
 39 			 e.preventDefault();
 40 		}else{
 41 			 e.returnValue = false;
 42 		};	
 43 	};
 44 	$.fn.drag = function(op){
 45 		var set = $.extend({
 46 			handle : null,
 47 			beforedrag : function(pos,self){},
 48 			ondraging : function(pos,self){},
 49 			enddrag : function(pos,self){},
 50 			unselectClass : "unselect",
 51 			xdrag : true,//容器left是否改变
 52 			ydrag : true//容器top是否改变	
 53 		},op);
 54 		return this.each(function(){
 55 			var t = $(this), _t = this, handle = set.handle === null ? t : t.find(set.handle), cachemousepos = {x : 0, y :0}, cachethispos = {x : 0, y: 0}, draggable = false, handled = false;
 56 			//获取鼠标位置
 57 			function getmousepos(e){
 58 				e = e || window.event;
 59 				return {
 60 					x : (function(){
 61 						return e.clientX + _window.scrollLeft();
 62 					})(),
 63 					y : (function(){
 64 						return e.clientY + _window.scrollTop();
 65 					})()
 66 				};
 67 			};
 68 			//获取容器的 left 和 top
 69 			function getdivpos(){
 70 				return {
 71 					x : (function(){
 72 						return t.css("left") === "auto" ? 0 : parseInt(t.css("left"))	
 73 					})(),
 74 					y : (function(){
 75 						return t.css("top") === "auto" ? 0 : parseInt(t.css("top"));
 76 					})()
 77 				};
 78 			};
 79 			//鼠标移动时 设置容器css positon left and top
 80 			function __mousemove(e){
 81 				e = e || window.event;
 82 				clsSelect(e);	
 83 				if(draggable){
 84 					handled = true;	
 85 					var eve = getmousepos(e);
 86 					var css = {};
 87 					if(set.xdrag === true){
 88 						css.left = eve.x - cachemousepos.x + cachethispos.x + "px";
 89 					};
 90 					if(set.ydrag === true){
 91 						css.top = eve.y - cachemousepos.y + cachethispos.y + "px"  
 92 					};
 93 					t.css(css);
 94 					 
 95 					set.ondraging(getdivpos(),t);
 96 				};
 97 			};
 98 			//鼠标停止松开后重置
 99 			function __mouseup(e){
100 				e = e || window.event;
101 				draggable  = false;	
102 				$(document).unbind("mousemove", __mousemove);
103 				$(document).unbind("selectstart", __mousemove);
104 				if(handled){
105 					set.enddrag(getdivpos(),t);
106 					handled = false;
107 				};
108 			};
109 			//初始化
110 			function init(){
111 				handle.mousedown(function(e){
112 					e = e || window.event;
113 					draggable = true;
114 					if($(this).attr("_ismaxed")){
115 						return;
116 					};
117 					cachethispos = getdivpos();
118 					cachemousepos = getmousepos(e);
119 					set.beforedrag(cachethispos,t);
120 					$(document).bind("mousemove", __mousemove).bind("mouseup", __mouseup);
121 					$(document).bind("selectstart", __mousemove); 
122 				});
123 			};
124 			init();
125 		});
126 	};
127 })(jQuery);
