137 		 * @returns {String} 转换结果字符串
138 		 */
139 		encode : function(object) {
140 			return (function() {
141 				var _this = this;
142 				if(_this===null || _this===undefined){
143 					return "" ;
144 				}
145 				if (_this instanceof Function || _this instanceof RegExp) {
146 					return _this.toString();
147 				}
148 				if (_this instanceof String) {
149 					return '"' + _this.toString() + '"';
150 				}
151 				if (_this instanceof Boolean || _this instanceof Number) {
152 					return _this.toString();
153 				}
154 				if (_this instanceof Date) {
155 					return _this.getTime();
156 				}
157 				if (_this instanceof Object || _this instanceof Array) {
158 					var tempArr = new Array;
159 					if (_this instanceof Array) {
160 						for ( var i = 0; i < _this.length; i++) {
161 							tempArr.push(arguments.callee.call(_this[i]));
162 						}
163 					} else {
164 						for ( var fieldname in _this) {
165 							if(_this[fieldname]==null || _this[fieldname]==undefined){
166 								tempArr.push(arguments.callee.call(fieldname) + ':' + '""');
167 								continue;
168 							}
169 							if (jQuery.type(_this[fieldname]) == 'function' || jQuery.type(_this[fieldname]) == 'regexp') {
170 								continue;
171 							}
172 							if (_this[fieldname] instanceof Object) {
173 								tempArr.push(arguments.callee.call(fieldname) + ':' + arguments.callee.call(_this[fieldname]));
174 								continue;
175 							}
176 							tempArr.push(arguments.callee.call(fieldname) + ':' + arguments.callee.call(_this[fieldname]));
177 						}
178 					}
179 					var retString = tempArr.join('');
180 					if (_this instanceof Array) {
181 						retString = '[' + tempArr.join(',') + ']';
182 					} else {
183 						retString = '{' + tempArr.join(',') + '}';
184 					}
185 					return retString;
186 				} else {
187 					return _this.toString();
188 				}
189 			}).call(object);
