 13 /**
 14  * @name Horn.Panel
 15  * @class
 16  * 面板组件</br>
 17  * 属于容器组件，一般做为单行组件的容器使用
 18  */
 19 /**@lends Horn.Panel# */
 20 /**
 21  * 组件的唯一标示
 22  * @name Horn.Panel#<b>id</b>
 23  * @type String
 24  * @default 
 25  */
 26 /**
 27  * 组件的名称
 28  * @name Horn.Panel#<b>name</b>
 29  * @type String
 30  * @default 
 31  */
 32 /**
 33  * 面板组件的标题栏,默认没有标题栏
 34  * @name Horn.Panel#<b>title</b>
 35  * @type String
 36  * @default ""
 37  */
 38 /**
 39  * 列数（目前支持1-4列）
 40  * @name Horn.Panel#cols
 41  * @type int
 42  * @default 3
 43  * @example
 44  * #@panel({"id":"testspanel","title":"","cols":3})
 45  * 
 46  * #end
 47  */
 48 /**
 49  * 是否启用展开/收缩功能,默认不启用
 50  * @name Horn.Panel#<b>expandable</b>
 51  * @type Boolean
 52  * @default false
 53  * @example
 54  * #@panel({"expandable":true})
 55  *	#textfield({"label":"流程名称：", "name":"hsBpmNameLike"})
 56  * #end
 57  */
 58 /**
 59  * 是否展开(只有在expandable为true生效 ),默认不展开
 60  * @name Horn.Panel#<b>isExpand</b>
 61  * @type Boolean
 62  * @default false
 63  * @example
 64  * #@panel({"expandable":true,"isExpand":true})
 65  *	#textfield({"label":"流程名称：", "name":"hsBpmNameLike"})
 66  * #end
 67  */
 68 	Horn.Panel = Horn.extend(Horn.Base,{
 69 		COMPONENT_CLASS:"Panel",
 70 		init:function(){
 71 		 Horn.Panel.superclass.init.apply(this,arguments);
 72 		 var expandDiv = this.el.children('div.m-panel-box');
 73 		 var _panel = this.el.children("div.g-wrap"); 
 74 		 if(expandDiv) {
 75 			 var btn = expandDiv.children().children("i");
 76 			 expandDiv.click(function(){
 77 				 if(btn.hasClass('fa-angle-up')){
 78 					_panel.hide();
 79 					btn.removeClass("fa-angle-up");
 80 					btn.addClass("fa-angle-down");
 81 				 }else{
 82 					 _panel.show();
 83 					btn.removeClass("fa-angle-down");
 84 					btn.addClass("fa-angle-up");
 85 				 }
 86 			}) ;
 87 		 }
 88 	},
 89         /**
 90          * 显示<br/>
 91          * 显示面板容器的内容
 92          * @name Horn.Panel#show
 93          * @function
 94          * @return {void}
 95          */
 96 		show : function(){
 97 			this.el.show();
 98 			if(this.params['expandable']==true){
 99 				this.closebar.show();
100 				this.expanedbar.hide();
101 			}
102 		},
103         /**
104          * 隐藏<br/>
105          * 隐藏面板容器及容器内的内容
106          * @name Horn.Panel#hide
107          * @function
108          * @return {void}
109          */
110 		hide : function(){
111 			this.el.hide();
112 			if(this.params['expandable']==true){
113 				this.closebar.hide();
114 				this.expanedbar.show();
115 			}
116 			
117 			
118 		},
119 		/**
120          * 设置内部输入组件为可用状态<br/>
121          * 对所有表单项起效（文本字段、文本区域、按钮、复选框、单选框和下拉框）
122          * @name Horn.Panel#enable
123          * @function
124          * @return {void}
125          */
126 		enable : function(){
127 			//this.el.find('input,textarea').not('input[realydisable],textarea[realydisable]').removeAttr('disabled');
128 			
129 			//BUG #6625 [panel]表单组件禁用启用
130 			Horn.Field.findFieldCompsIn(this.el).each(function(i,f){
131 	    		this.setDisabled(false);
132 	    	});
133 			//BUG #6626 [panel]禁用表单类组件
134 			this.el.find('button,select').removeAttr('disabled');
135 		},
136 		/**
137          * 设置内部输入组件为不可用状态(disabled)<br/>
138          * 对所有表单项起效（文本字段、文本区域、按钮、复选框、单选框和下拉框）
139          * @name Horn.Panel#disable
140          * @function
141          * @return {void}
142          */
143 		disable : function(){
144 			//this.el.find('input[disabled],textarea[disabled]').attr('realydisable','yes');
145 			//this.el.find('input,textarea').not('input[realydisable],textarea[realydisable]').attr('disabled','disabled');
146 			
147 			//BUG #6625 [panel]表单组件禁用启用
148 			Horn.Field.findFieldCompsIn(this.el).each(function(i,f){
149 	    		this.setDisabled(true);
150 	    	});
151 			
152 			//BUG #6626 [panel]禁用表单类组件
153 			this.el.find('button,select').removeAttr('disabled');
154 			this.el.find('button,select').attr('disabled','disabled');
155 		}
156 	});
157 	Horn.regUI('div.g-panel',Horn.Panel);
