  7 /**
  8  * @description 可绑定按钮的多级菜单
  9  * @name Horn.Menu
 10  * @class
 11  * 可绑定按钮的多级菜单
 12  * @extends Horn.Base
 13  * @example
 14     #menu({"name":"text11","opendir":"R","items":[{"label":"test","hasLine":true,"icon":"http://www.baidu.com/img/bdlogo.png","items":[{"label":"test"},{"label":"test","items":[{"label":"test"},{"label":"test"},{"label":"test"}]},{"label":"test"}]},{"label":"test"},{"label":"test"}]})
 15  */
 16 /**
 17  * 组件的唯一标示
 18  * @name Horn.Menu#<b>id</b>
 19  * @type String
 20  * @default ""
 21  * @example
 22  * 无
 23  */
 24 /**
 25  * 表单输入域的名称
 26  * @name Horn.Menu#<b>name</b>
 27  * @type String
 28  * @default ""
 29  * @example
 30  * 无
 31  */
 32 /**
 33  * 表单输入域的别名，用在表单中存在相同name的情况下，可以通过别名来区分
 34  * @name Horn.Menu#<b>alias</b>
 35  * @type String
 36  * @default ""
 37  * @example
 38  * 无
 39  */
 40 /**
 41  * 菜单的展开方向,向右R,向左L；当menu位于窗口最右的时候向左展开，当其位于窗口最左的时候向右展开
 42  * @name Horn.Menu#<b>opendir</b>
 43  * @type String
 44  * @default "R"
 45  * @ignore
 46  * @example
 47  * 无
 48  */
 49 /**
 50  * 菜单的条目内容，其中主要的配置内容为label：显示名；icon：图标；event:点击的处理事件；items：子级菜单；tips：显示的提示信息
 51  * @name Horn.Menu#<b>items</b>
 52  * @type String
 53  * @default "1"
 54  * @example
 55  * 无
 56  */
 57 Horn.Menu = Horn.extend(Horn.Base,{
 58 		COMPONENT_CLASS:"menu",
 59 		id:"menu",
 60 		baseMenuEl:null,
 61 		baseMenuItem:null,
 62 		rootMenu:null,
 63 		zindexInit:9999,
 64 		opendir:"R",//展开方向，默认向右展开
 65 		tier:0,//menu菜单有多少层
 66 		classOneNum:0,//一级菜单的个数
 67 		init:function(dom){
 68 			 Horn.Menu.superclass.init.apply(this,arguments);
 69 			 var _this = this;
 70 			 
 71 			 this.classOneNum = this.params.items.length;//获取一级菜单的个数
 72 			 
 73 			 this.id = this.params.id?this.params.id:this.id;
 74 			 this.el.attr("id",this.id);
 75 			 this.opendir = this.params.opendir?this.params.opendir:this.opendir;
 76 			 this.baseMenuEl = this.el.next();
 77 			 this.baseMenuItem = $("li",this.baseMenuEl);
 78 			 this.recursionItems(this.baseMenuEl,this.params)
 79 			 this.baseMenuEl.appendTo(this.el);
 80 			 $("li[base='']",this.el).remove();//去除显示的基本元素，IE下如果不写=‘’竟然找不到节点
 81 			 
 82 			 this.rootMenu = this.el.children("ul");
 83 			 
 84 			 //在datagrid中绑定menu时候datagrid的层高过高导致menu被遮盖
 85 			 this.rootMenu.css({"z-index":_this.zindexInit-1});
 86 			 
 87 			 $(document).bind("click",function(e){
 88 				 //不管能不能找到都不能让事件再冒泡了，
 89 				 if($(e.target).closest("*[refmenu="+_this.id+"]").size() == 0 && e.button == 0){
 90 					 _this.hideMenu();
 91 					 //return false;//阻止冒泡导致radio无法切换，checkbox无法选中
 92 				 }
 93 			 })
 94 			 
 95 			 if(this.params.isContextMenu){
 96 				 document.oncontextmenu=function(ev){
 97 					ev = ev || window.event;  
 98 					if(ev.button == 2 || ev.type == "contextmenu"){
 99 						var explorer =navigator.userAgent ;
100 						//ie 
101 						if (navigator.appName!= 'Microsoft Internet Explorer'){
102 							window.event = ev;
103 						}
104 						_this.showMenu();
105 						return false;
106 					}
107 				 }
108 			 }
109 			 
110 			 //菜单主动将自己绑定到制定的元素上面
111 			 $("*[refmenu="+_this.id+"]").click(function(){
112 				 //增加被绑定元素是否禁用的判断，如果被禁用就不弹出菜单
113 				 if(!_this.isContextDisabled($(this))){
114 					 _this.reCalOpendir($(this));
115 					 if (!_this.rootMenu.is(":visible")) {
116 						 _this.showMenu($(this));
117 					 } else {
118 						 _this.hideMenu();
119 					 }
120 				 }
121 			 });
122 			 
123 			 //改变被绑定按钮的展示样式
124 			 $("*[refmenu="+_this.id+"]").each(function(i,o){
125 				 $("<i class='fa fa-caret-down'></i>").appendTo($(o));
126 				 if(o.tagName == "A" || o.tagName == "a"){
127 					 $(o).css({"text-decoration":"none"})
128 				 }
129 				 //不再兼容IE7
130 //				 if(o.tagName == "BUTTON" || o.tagName == "button" && $.browser.msie && $.browser.version=="7.0"){
131 //					 $(o).css({"height":"25px"})
132 //				 }
133 			 })
134 			 
135 			//页面resize的时候onresize
136 			 $(window).bind("resize",function(){
137 				_this.hideMenu(); 
138 			 });
139 			 
140 			 //修复当菜单不是最先被初始化的时候在菜单和按钮之间会出现很大的空隙
141 			 this.el.prependTo($(document.body));
142 		},
143 		reCalOpendir:function(context){//重新计算展开方向
144 			var maxWidth ;
145 			var p = {};
146 			var toLeft = context.offset().left;
147 			var itemWidth = 150;
148 			if(this.tier < 4){//可以保证三层以内的自动判断，不回折
149 				maxWidth = this.tier*itemWidth
150 				p.left = maxWidth < toLeft+context.width();
151 				p.right = maxWidth < $(window).width() - toLeft;
152 			}else{
153 //				maxWidth = 3*itemWidth
154 //				p.left = maxWidth < toLeft-context.width();
155 //				p.right = maxWidth < $(window).width() - toLeft;
156 				p.left = p.right = true;
157 			}
158 			if(p.left && p.right || (!p.left && !p.right)){//如果向左
159 				return;
160 			}
161 			if(this.opendir == "L" && !p.left) this.opendir = "R";
162 			if(this.opendir == "R" && !p.right) this.opendir = "L";
163 		},
164 		recursionItems:function(menuEl,item){
165 			this.tier++;
166 			var _this = this;
167 			if(item.items){
168 				var i = 0;
169 				while(i < item.items.length){
170 					
171 					var tempItem = _this.processparam(_this.baseMenuItem.clone(true).removeAttr("base"),item.items[i]);
172 					if(item.items[i].items){
173 						_this.zindexInit++;
174 						var nextEl = _this.baseMenuEl.clone(true).css({"z-index":_this.zindexInit});
175 //						var nextEl = _this.baseMenuEl.clone(true);
176 						nextEl.appendTo(tempItem)
177 						_this.recursionItems(nextEl,item.items[i]);
178 					}
179 					tempItem.appendTo(menuEl);
180 					
181 					i++;
182 				}
183 			}
184 		},
185 		processparam:function(menuEl,params){
186 			if(params.tips)
187 				$("a",menuEl).attr("title",params.tips)
188 			if(params.hasLine)
189 				$("span.h_toolbar-item-txt",menuEl).addClass("h_menu-line");
190 			if(params.event)
191 				$("a",menuEl).attr("onclick",params.event)
192 			if(params.icon)
193 				$("span.h_toolbar-item-icon",menuEl).html("<img width='20px' src='"+params.icon+"'>");
194 			if(params.label)
195 				$("span.h_toolbar-item-txt",menuEl).text(params.label);
196 			if(!params.items)
197 				$("span.h_toolbar-item-arrow",menuEl).remove()
198 			return menuEl;
199 		},
200 		/**
201 		 * 在哪个元素上进行绝对定位
202 		 */
203 		showMenu:function(context,el,isUl){
204 			var _this = this;
205 			if(!el){
206 				el = this.rootMenu;
207 			}
208 			
209 			var pos = this.getShowPosition(context,isUl);
210 			el.css({"top":pos.Y,"left":pos.X})
211 			
212 			el.show();
213 			
214 			el.children("li").hover(function(){
215 				var child = $(this).children("ul");
216 				if(child.size() == 1){
217 					_this.showMenu($(this),child,true);
218 				}
219 			},function(){
220 				var child = $(this).children("ul");
221 				if(child.size() == 1){
222 					child.hide()
223 				}
224 			})
225 		},
226 		/**
227 		 * 获取菜单应该显示在的位置
228 		 */
229 		getShowPosition:function(context,isUl){
230 			var _this = this;
231 			var itemWidth = 150;//菜单条目的宽度
232 			var toLeft,toTop,width,height;
233 			
234 			if(!context){//页面自定义右键菜单
235 				toLeft = window.event.clientX;
236 				toTop = window.event.clientY;
237 				height=0;
238 				width=0;
239 			}else{//被绑定菜单，context是被绑定元素的jquery对象
240 				toLeft = context.offset().left;
241 				toTop = context.offset().top;
242 				width = context.outerWidth();
243 				height = context.outerHeight();
244 			}
245 			/*****************以上属性对以下函数只读******************************/		
246 			var ulPos = function(){
247 				var p_y,p_x; //菜单显示的坐标点
248 				p_y=-height;
249 				if (_this.opendir == "R"&& (toLeft + width + itemWidth) > $(window).width() || _this.opendir == "L") {//要么都向左展开，要么都向右展开
250 					_this.opendir = "L";
251 					p_x=-width;
252 				}
253 				if(_this.opendir == "L" && (toLeft-itemWidth) < 0 || _this.opendir == "R"){
254 					_this.opendir = "R";
255 					p_x=width;
256 				}
257 				return {"X":p_x,"Y":p_y};
258 			}
259 			
260 			var contextPos = function(){
261 				var p_y,p_x; //菜单显示的坐标点
262 				var num = _this.classOneNum;
263 				
264 				//STORY #10125 #10019建议menu要能支持浏览器向上或是向下碰撞 
265 				//判断被绑定的按钮是不是在页面视口的底部
266 				var w_h = $(window).height();
267 				var m_h = num * 39;//每个菜单的宽度是39
268 				if(toTop + width + m_h > w_h){
269 					p_y = toTop-m_h;
270 				}else{
271 					p_y = toTop + height;
272 				}
273 				
274 				//在这里判断从左边缘还是右边缘显示菜单
275 				if((toLeft+itemWidth)>$(window).width() || (_this.opendir == "L" && (toLeft-itemWidth) > 0)){
276 					p_x = toLeft-itemWidth+width;//这里的26px是下拉箭头的的i元素的宽度
277 					_this.opendir = "L";
278 				}else
279 					p_x = toLeft;
280 				return {"X":p_x,"Y":p_y};
281 			}
282 			/*************************以上获取位置的函数定义，以下处理位置******************************************/
283 			if(isUl){//子级菜单
284 				return ulPos();
285 			}else{//父菜单或者主菜单
286 				return contextPos();
287 			}
288 		},
289 		hideMenu:function(){
290 			this.rootMenu.hide();
291 			this.rootMenu.css({"top":1200,"left":600})
292 			this.opendir=this.params.opendir;
293 		},
294 		/**
295 		 * @description 添加条目的处理，现在只能添加到指定一级菜单的末尾
296 	     * @function
297 	     * @name Horn.Menu#addItem
298 	     * @param {object} obj
299 	     * @ignore
300 	     */
301 		addItem:function(obj){
302 			var tempItem = this.processparam(this.baseMenuItem.clone(true).removeAttr("base"),obj);
303 			tempItem.appendTo(this.rootMenu)
304 		},
305 		isContextDisabled:function(context){
306 			return context.hasClass("h_btn-disabled");
307 		}
308 	});
309 Horn.regUI('div.hc_menu',Horn.Menu);
