 88 	Horn.HiddenField = Horn.extend(Horn.Field,{
 89 		COMPONENT_CLASS:"HiddenField",
 90 		init : function(){
 91 			Horn.HiddenField.superclass.init.apply(this,arguments);
 92 			this.field = this.el;
 93 			this.name = this.field.attr("name") ;
 94             this.alias = this.field.attr("alias") || "" ;
 95             this.defValue = (this.params.defValue != undefined)?this.params.defValue:this.field.val();
 96 		},
 97 		isValid : function(){
 98 			return true;
 99 		},
100 		showError : function(){
101 		},
102 		reset : function(clear){
103 			if(clear) {
104                 this.field.val("");
105             }else{
106                 this.field.val(this.defValue);
107             }
108 		},
109 		removeError : function(){
110 		}
111 	}); 
112 	Horn.Field.regFieldType("input.hc_hiddenfield",Horn.HiddenField) ;
